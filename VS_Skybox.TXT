cbuffer staticMeshBuffer : register(b0)
{
    float4x4 W;
    float4x4 VP;
};

struct VS_IN {
    float3 Pos : POSITION;
    float3 Normal : NORMAL;
    float3 Tangent : TANGENT;
    float2 UV : TEXCOORD;
};

struct PS_IN {
    float4 Pos : SV_POSITION;
    float2 UV  : TEXCOORD;
};

PS_IN VS(VS_IN input)
{
    PS_IN o;
    float4 pos = float4(input.Pos, 1.f);

    pos = mul(pos, W);
    pos = mul(pos, VP);

    o.Pos = pos;
    o.UV = input.UV;
    return o;
}
