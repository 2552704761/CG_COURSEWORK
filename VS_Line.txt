cbuffer Camera : register(b0)
{
    float4x4 VP;
};

struct VS_IN
{
    float3 pos   : POSITION;
    float3 color : COLOR;
};

struct VS_OUT
{
    float4 pos   : SV_POSITION;
    float3 color : COLOR;
};

VS_OUT VS(VS_IN input)
{
    VS_OUT o;
    o.pos = mul(float4(input.pos, 1.0f), VP);
    o.color = input.color;
    return o;
}
