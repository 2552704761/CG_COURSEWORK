cbuffer staticMeshBuffer : register(b0)
{
    float4x4 W;
    float4x4 VP;
    float time;
};

struct VS_INPUT
{
    float4 Pos : POSITION;
    float3 Normal : NORMAL;
    float3 Tangent : TANGENT;
    float2 TexCoords : TEXCOORD;
    float4x4 World : WORLD;
};

struct PS_INPUT
{
    float4 Pos : SV_POSITION;
    float3 Normal : NORMAL;
    float3 Tangent : TANGENT;
    float2 TexCoords : TEXCOORD;
};

PS_INPUT VS(VS_INPUT input)
{
    PS_INPUT output;

    output.Pos = mul(input.Pos, input.World);
    float windStrength = 0.2f;
    float windSpeed    = 2.0f;
    float heightFactor = saturate(input.Pos.y);
    float windOffset =
        sin(time * windSpeed + output.Pos.x * 0.5f) *
        windStrength *
        heightFactor;
    output.Pos.xz += windOffset * float2(1, 0.3);
    output.Pos = mul(output.Pos, VP);
    output.Normal = mul(input.Normal, input.World);
    output.Tangent = mul(input.Tangent, input.World);
    output.TexCoords = input.TexCoords;
    return output;
}
